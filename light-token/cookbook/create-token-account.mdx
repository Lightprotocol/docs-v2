use light_token_sdk::token::CreateTokenAccountCpi;

let account_cpi = CreateTokenAccountCpi {

    mint: mint.clone(),
    owner: data.owner,
    compressible: Some(compressible_params),
};

let signer_seeds: &[&[u8]] = &[TOKEN_ACCOUNT_SEED, &[bump]];
