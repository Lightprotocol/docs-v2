1. On-ramps need to create ATAs for every user's to hold a specific token. Light tokens reduce this cost by 200x.
2. Your users receive the same token, just stored more efficiently.
3. The light-token API matches the SPL-token API almost entirely, 
and extends their functionality to include the light token program in addition to the SPL-token and Token-2022 programs. 

| Creation Cost     | SPL                 | light-token          |
| :---------------- | :------------------ | :------------------- |
| **Token Account** | ~2,000,000 lamports | ~**11,000** lamports |

## On-Ramp Transfer

1. Derives the user's light-token ATA
2. Creates the ATA (idempotent)
3. Transfers tokens from hot wallet to user

<Note>
Find full code examples [here](https://github.com/Lightprotocol/light-token-toolkits).
</Note>

<Tabs>
<Tab title="Action">

```typescript
import {
  getOrCreateAtaInterface,
  getAssociatedTokenAddressInterface,
  transferInterface,
} from "@lightprotocol/compressed-token/unified";

async function onrampTransfer(
  rpc: Rpc,
  hotWallet: Keypair,
  userWallet: PublicKey,
  mint: PublicKey,
  amount: bigint
) {
  // Create user's ATA if needed
  await getOrCreateAtaInterface(rpc, hotWallet, mint, userWallet);

  // Transfer from hot wallet to user
  const hotWalletAta = getAssociatedTokenAddressInterface(mint, hotWallet.publicKey);
  const userAta = getAssociatedTokenAddressInterface(mint, userWallet);

  return transferInterface(rpc, hotWallet, hotWalletAta, mint, userAta, hotWallet, amount);
}
```

<Accordion title="Compare to SPL">

```typescript
import {
  getOrCreateAssociatedTokenAccount,
  getAssociatedTokenAddressSync,
  transfer,
} from "@solana/spl-token";

async function onrampTransferSPL(
  connection: Connection,
  hotWallet: Keypair,
  userWallet: PublicKey,
  mint: PublicKey,
  amount: bigint
) {
  // Create user's ATA if needed
  await getOrCreateAssociatedTokenAccount(connection, hotWallet, mint, userWallet);

  // Transfer from hot wallet to user
  const hotWalletAta = getAssociatedTokenAddressSync(mint, hotWallet.publicKey);
  const userAta = getAssociatedTokenAddressSync(mint, userWallet);

  return transfer(connection, hotWallet, hotWalletAta, userAta, hotWallet, amount);
}
```

<Tab title="Instruction">

```typescript
import { Transaction } from "@solana/web3.js";
import {
  getAssociatedTokenAddressInterface,
  createAssociatedTokenAccountInterfaceIdempotentInstruction,
  createLoadAtaInstructions,
  createTransferInterfaceInstruction,
} from "@lightprotocol/compressed-token/unified";
import { CTOKEN_PROGRAM_ID } from "@lightprotocol/stateless.js";

async function buildOnrampTx(
  rpc: Rpc,
  hotWallet: PublicKey,
  userWallet: PublicKey,
  mint: PublicKey,
  amount: bigint
) {
  const hotWalletAta = getAssociatedTokenAddressInterface(mint, hotWallet);
  const userAta = getAssociatedTokenAddressInterface(mint, userWallet);

  const tx = new Transaction().add(
    // Create user's ATA (idempotent)
    createAssociatedTokenAccountInterfaceIdempotentInstruction(
      hotWallet,
      userAta,
      userWallet,
      mint,
      CTOKEN_PROGRAM_ID
    ),
    // Load hot wallet's ATA
    ...(await createLoadAtaInstructions(rpc, hotWalletAta, hotWallet, mint, hotWallet)),
    // Transfer to user
    createTransferInterfaceInstruction(hotWalletAta, userAta, hotWallet, amount)
  );

  return tx;
}
```

<Accordion title="Compare to SPL">

```typescript
import {
  getAssociatedTokenAddressSync,
  createAssociatedTokenAccountIdempotentInstruction,
  createTransferInstruction,
} from "@solana/spl-token";

function buildOnrampTxSPL(

  userWallet: PublicKey,
  mint: PublicKey,

  return tx;
}
```

## Exit to SPL

When users withdraw to an integration that does not support light-token yet,
you'll need to unwrap light-token to SPL.

Unwrap moves tokens from light-token ATA to SPL/Token-2022 account.

<Note>
Find a full code example [here](https://github.com/Lightprotocol/examples-light-token/blob/main/toolkits/payments-and-wallets/unwrap.ts).
</Note>

<Tabs>
<Tab title="Action">

```typescript
import { getAssociatedTokenAddressSync } from "@solana/spl-token";

// SPL ATA must exist
const splAta = getAssociatedTokenAddressSync(mint, owner.publicKey);

await unwrap(rpc, payer, splAta, owner, mint, amount);
```

</Tab>
<Tab title="Instruction">

```typescript
import { Transaction } from "@solana/web3.js";
import { getAssociatedTokenAddressSync } from "@solana/spl-token";
import {
  createLoadAtaInstructions,
  createUnwrapInstruction,
  getAssociatedTokenAddressInterface,
} from "@lightprotocol/compressed-token/unified";
import { getSplInterfaceInfos } from "@lightprotocol/compressed-token";

const ctokenAta = getAssociatedTokenAddressInterface(mint, owner.publicKey);
const splAta = getAssociatedTokenAddressSync(mint, owner.publicKey);

const splInterfaceInfos = await getSplInterfaceInfos(rpc, mint);
const splInterfaceInfo = splInterfaceInfos.find((i) => i.isInitialized);

const tx = new Transaction().add(
  ...(await createLoadAtaInstructions(
    rpc,
    ctokenAta,
    owner.publicKey,
    mint,
    payer.publicKey
  )),
  createUnwrapInstruction(
    ctokenAta,
    splAta,
    owner.publicKey,
    mint,
    amount,
    splInterfaceInfo
  )
);
```

## Entry from SPL

Wrap tokens from SPL/Token-2022 accounts to light-token ATA.

<Note>
Find a full code example [here](https://github.com/Lightprotocol/examples-light-token/blob/main/toolkits/payments-and-wallets/wrap.ts).
</Note>

<Tabs>
<Tab title="Action">

```typescript
import { getAssociatedTokenAddressSync } from "@solana/spl-token";
import {
  wrap,
  getAssociatedTokenAddressInterface,
} from "@lightprotocol/compressed-token/unified";

// SPL ATA with tokens to wrap
const splAta = getAssociatedTokenAddressSync(mint, owner.publicKey);
// c-token ATA destination
const ctokenAta = getAssociatedTokenAddressInterface(mint, owner.publicKey);

await wrap(rpc, payer, splAta, ctokenAta, owner, mint, amount);
```

</Tab>
<Tab title="Instruction">

```typescript
import { Transaction } from "@solana/web3.js";
import { getAssociatedTokenAddressSync } from "@solana/spl-token";
import {
  createWrapInstruction,
  getAssociatedTokenAddressInterface,
} from "@lightprotocol/compressed-token/unified";
import { getSplInterfaceInfos } from "@lightprotocol/compressed-token";

const ctokenAta = getAssociatedTokenAddressInterface(mint, owner.publicKey);

const splInterfaceInfos = await getSplInterfaceInfos(rpc, mint);
const splInterfaceInfo = splInterfaceInfos.find((i) => i.isInitialized);

const tx = new Transaction().add(
  createWrapInstruction(
    splAta,
    ctokenAta,
    owner.publicKey,
    mint,
    amount,
    splInterfaceInfo
  )
);
```
