```rust
use light_token_client::actions::{CreateAta, CreateMint};
use rust_client::setup_rpc_and_payer;
use solana_sdk::{signature::Keypair, signer::Signer};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let (mut rpc, payer) = setup_rpc_and_payer().await;
    let mint_seed = Keypair::new();

    // Create mint
    let mint_result = CreateMint {
        decimals: 9,
        freeze_authority: None,
        token_metadata: None,
    }
    .execute(&mut rpc, &payer, &mint_seed)
    .await?;

    // Create ATA
    let ata = CreateAta {
        mint: mint_result.mint,
        owner: payer.pubkey(),
        idempotent: true,
    }
    .execute(&mut rpc, &payer)
    .await?;

    println!("ATA: {ata}");

    Ok(())
}
```
